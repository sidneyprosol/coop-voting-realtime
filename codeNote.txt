    <script>
        let votes = {};

        function selectCandidate(card, dept) {
            let group = card.closest('.department-box').querySelectorAll('.candidate-card');
            group.forEach(c => c.classList.remove('selected'));

            card.classList.add('selected');
            votes[dept] = card.querySelector('.candidate-name').innerText;
        }

        function submitVote() {
            console.log(votes);
            alert("Vote Submitted!\n" + JSON.stringify(votes, null, 2));
        }
